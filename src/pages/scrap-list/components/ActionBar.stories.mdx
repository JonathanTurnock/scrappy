# Desktop Action Bar

Desktop Action Bar for the Scrap List

export const DesktopTemplate = ({ selectionCount }) => (
  <Stack style={{ height: "500px" }}>
    <DesktopActionBar
      selectionCount={selectionCount}
      onAdd={() => alert("Add Clicked")}
      onRename={() => alert("Rename Clicked")}
      onDelete={() => alert("Delete Clicked")}
      onToggleSelection={() => alert("Selection Mode Changed")}
      onFilter={(filterText) => alert(`Filter By ${filterText}`)}
    />
  </Stack>
)

<Canvas>
  <Story name="DesktopActionBar" args={{ selectionCount: 1 }}>
    {DesktopTemplate.bind({})}
  </Story>
</Canvas>
